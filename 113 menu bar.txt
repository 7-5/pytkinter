from tkinter import *

# menu bar

def functions():

def functional():

window = Tk()

menubar= Menu(window)
window.config(menu=menubar)

filemenu= Menu(menubar)
menubar.add_cascade(label="File",menu=filemenu,tearoff=0) -> dropdown is cascade
filemenu.add_command(label="Open",command=functions)
filemenu.add_command(label="Save",command=functional)
filemenu.add_separator() # -----------------
filemenu.add_command(label="Exit",command=quit)

editmenu= Menu(menubar)
menubar.add_cascade(label="Edit",menu=editmenu,tearoff=0) -> dropdown is cascade
editmenu.add_command(label="Copy",command=functional)

window.mainloop()
