from tkinter import *

#  grid()= geometry mgr that organises widgets in a table like structure in a parent
# row zero column zero. that sort of shit. 

window = Tk()

name=Label(window,text="name: ").grid(row=0,column=0)
nameentry= Entry(window).pack().grid(row=0,column=1)

surname=Label(window,text="name: ").grid(row=1,column=0)
surnameentry= Entry(window).pack().grid(row=1,column=1)

email=Label(window,text="name: ").grid(row=2,column=0)
emailentry= Entry(window).pack().grid(row=2,column=1)

submitbutton=Button(window,text="submit").grid(row=3,column=0,columnspan=2)


window.mainloop()
