from tkinter import *

# Entry widget = textbox that accepts single line of user input


def submit():
	#submit func
	name=entry.get()
	print("hi "+name)
	#delete func
	entry.delete(0,END) # delete all
	#backspace
	entry.delete(END-1,END) # delete all
	entry.config(state=DISABLED) # key to updating changes like a counter maybe.
	
window = Tk()  
 
entry = Entry(window,
# all the other label customizer stuff. even images.
show="*" #passwords
)

entry.insert(0,"spongebobg")
entry.pack(side=LEFT)

submit_button= Button(window,
text="sumit",
command=submit,
state=ACTIVE/DISABLED,
)
submit_button.pack(side=RIGHT)

window.mainloop()
